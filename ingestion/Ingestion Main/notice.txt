def clean_markdown(raw_text: str, image_dict) -> str:
    print("3. Đang dọn dẹp (Cleaning) mã Markdown...")
    for name in image_dict.keys():
        raw_text = raw_text.replace(f"![Ngữ cảnh: Hình ảnh được trích xuất từ {name}]", "") 
        raw_text = raw_text.replace(f"![Lỗi trích xuất ảnh {name}]", "")
    text = re.sub(r'\n{3,}', '\n\n', raw_text)
    - re.sub(r'\n{3,}', '\n\n', raw_text) sẽ thay thế tất cả các chuỗi gồm ba hoặc nhiều ký tự xuống dòng liên tiếp bằng chỉ hai ký tự xuống dòng. Điều này giúp làm sạch văn bản bằng cách loại bỏ các khoảng trắng thừa và đảm bảo rằng văn bản có định dạng gọn gàng hơn.
    return text.strip()

    def chunk_for_rag(cleaned_text: str) -> list:
    print("4. Đang cắt nhỏ dữ liệu (Chunking)...")
    - separators là một list các ký tự xác định để cắt nhỏ văn bản cho cấu trúc markdown, các separators được sắp xếp theo thứ tự ưu tiên từ cao đến thấp, giúp đảm bảo rằng các đoạn văn bản được cắt một cách hợp lý và giữ nguyên cấu trúc ý nghĩa của nội dung. Cụ thể:
    separators = ["\n#{1, 6}"(header 1 --> 6), "\n\n", "\n", " ", ""]
    - sử dụng RecursiveCharacterTextSplitter để cắt nhỏ văn bản dựa trên các ký tự phân tách đã định nghĩa trong danh sách separators. Các đoạn văn bản sẽ có kích thước tối đa là 1500 ký tự và có sự chồng lấn giữa các đoạn là 150 ký tự để đảm bảo rằng các ý nghĩa không bị mất mát khi cắt nhỏ.
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=1500,
        chunk_overlap=150,
        separators=separators,
        keep_separator=True
    )
    return splitter.split_text(cleaned_text)
